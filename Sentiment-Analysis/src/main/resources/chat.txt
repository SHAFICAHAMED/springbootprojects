                double borrowCost = (book.originalPrice / 100) * days;
                student.balance -= borrowCost;

                double vendorEarnings = borrowCost * 0.8;
                double libraryEarningsShare = borrowCost * 0.2;
                libraryEarnings += libraryEarningsShare;

                vendors.get(book.vendorCode).profit += vendorEarnings;
                book.currentPrice -= borrowCost / 2;
                book.status = "Lend";
                book.responsible = studentCode;
                System.out.println("Borrow successful. Remaining balance: " + student.balance);
            } else {
                System.out.println("Borrow failed. Check balance or book availability.");
            }
        } else if (option.equalsIgnoreCase("Check Balance")) {
            System.out.print("Enter Student Code: ");
            String studentCode = scanner.nextLine();
            Student student = students.get(studentCode);
            if (student != null) {
                System.out.println("Balance: " + student.balance);
            } else {
                System.out.println("Student not found.");
            }
        } else if (!option.equalsIgnoreCase("Exit")) {
            System.out.println("Invalid option. Try again.");
        }
    }

    private static void browseBooks() {
        System.out.println("Code | V Code | Name        | Author | OG Price | Curr Price | Status | Resp");
        for (Book book : books) {
            System.out.printf("%s    | %s     | %-10s | %-6s | %.2f    | %.2f     | %-8s | %s\n",
                    book.code, book.vendorCode, book.name, book.author,
                    book.originalPrice, book.currentPrice, book.status, book.responsible);
        }
    }

    private static void viewAccounts() {
        System.out.println("Total Library Earnings: " + libraryEarnings);
        System.out.println("Vendor Profits:");
        for (Vendor vendor : vendors.values()) {
            System.out.printf("Vendor Code: %s, Name: %s, Profit: %.2f\n", vendor.code, vendor.name, vendor.profit);
        }
    }

    private static Book findBookByCode(String code) {
        for (Book book : books) {
            if (book.code.equals(code)) {
                return book;
            }
        }
        return null;
    }
